```python
"""FND-04 + BCK-02: CI-blocking no-look-ahead mutation test.

Integration test. Calls the REAL screener.backtest.vbt_runner.run() with synthetic
OHLCV (via monkeypatch) and a perfect-foresight signal (signal[t] = close[t+1] > close[t]).

Assertion thresholds (per CONTEXT.md D-07 REVISED 2026-05-16; 05-RESEARCH.md §B Q5):
  - _lookahead=False  ->  abs(total_return) < 0.50  (correct path; .shift(1) negates foresight)
  - _lookahead=True   ->  total_return > 1.00       (mutation backdoor; foresight wins)

Threshold rationale: 10-seed Monte Carlo (RESEARCH §B Q5) on 250-bar mean-zero
GBM showed the original D-07 wording "<= 2x BH" is seed-sensitive because
shifted-foresight earns ~25% on near-zero-drift GBM via autocorrelation. The
0.50 / 1.00 absolute thresholds gave 4x separation across all 10 tested seeds
with seed=42 + loc=0.0.

B-2 fix (iter 2): The fixture span was extended from 250 bars × 1 ticker to
1008 bars × 3 tickers (≈4 calendar years) so vbt_runner.run() produces ≥1
complete 3yr-IS / 1yr-OOS window. The 0.50 / 1.00 thresholds were CALIBRATED
on the 250-bar fixture; the planning-time assumption is they remain valid on
the 1008-bar fixture because:
  (a) The market-vs-strategy mechanic is unchanged — mean-zero GBM still
      destroys the autocorrelation edge after .shift(1).
  (b) A longer series gives the perfect-foresight strategy MORE compounding
      opportunities, which strengthens (not weakens) the > 1.00 separation.
  (c) Mean-zero drift on a longer GBM has roughly the same |total_return|
      distribution (close to 0); 50% remains a comfortable noise ceiling.
If during execution either threshold proves non-robust on the 1008-bar fixture
(test fails on the first GREEN run), the executor must:
  1. Log a DEVIATION in the SUMMARY (NOT silently relax the threshold).
  2. Run a 10-seed Monte Carlo on the new fixture span to recalibrate.
  3. Ping the user to update D-07 (REVISED-2) before committing.

Mutation check (manual): removing `.shift(1, fill_value=False)` from
`vbt_runner.run()`'s else-branch is equivalent to hardcoding `_lookahead=True` —
test_no_lookahead_correct_path will fail. This is the FND-04 CI gate.
"""

from __future__ import annotations

from typing import Any

import pandas as pd
import pytest

from screener.backtest import vbt_runner

LOOKAHEAD_FALSE_MAX_RETURN = 0.50  # 50% — correct path must stay near random
LOOKAHEAD_TRUE_MIN_RETURN = 1.00   # 100% — mutation must produce wild outperformance


def _fixture_date_range(panel: pd.DataFrame) -> tuple[str, str]:
    """Derive (start, end) ISO date strings from the fixture's date index.

    B-2 fix (iter 2): test date range MUST cover ≥4 calendar years so
    vbt_runner.run() produces ≥1 complete 3yr-IS / 1yr-OOS window. The
    synthetic_ohlcv_panel fixture is calibrated to span 1008 bdays (≈4yr);
    we read its endpoints rather than hardcoding dates, so any future
    fixture-span change propagates here automatically.
    """
    dates = panel.index.get_level_values("date")
    return (
        dates.min().strftime("%Y-%m-%d"),
        dates.max().strftime("%Y-%m-%d"),
    )


def _build_perfect_foresight_snapshot(panel: pd.DataFrame) -> pd.DataFrame:
    """Construct a fake snapshot DataFrame matching what _load_snapshots_in_range returns.

    Perfect-foresight signal per D-06: True at bar t when close[t+1] > close[t].
    Last bar is False (no next bar exists).

    Schema must match the columns vbt_runner.run() consumes:
      - date (pd.Timestamp), ticker (str), passes_trend_template (bool),
        regime_state (str), composite_score (float)
    """
    # panel is MultiIndex (ticker, date). Extract per-ticker close, build signal.
    close_wide = panel["close"].unstack(level="ticker")  # date x ticker
    foresight = (close_wide.shift(-1) > close_wide).astype(bool)
    foresight = foresight.fillna(False)  # last bar has no next-day -> False

    # Stack back to long format with the required columns.
    rows: list[dict[str, Any]] = []
    for date in foresight.index:
        for ticker in foresight.columns:
            rows.append(
                {
                    "date": date,
                    "ticker": ticker,
                    "passes_trend_template": bool(foresight.loc[date, ticker]),
                    "regime_state": "Confirmed Uptrend",
                    "composite_score": 50.0,
                    "regime_score": 1.0,
                }
            )
    return pd.DataFrame(rows)


@pytest.fixture
def _patched_runner(
    monkeypatch: pytest.MonkeyPatch, synthetic_ohlcv_panel: pd.DataFrame
) -> tuple[str, str]:
    """Monkeypatch vbt_runner's data-loading seams to return synthetic data.

    Per RESEARCH §B Q7: patch the SYMBOL inside vbt_runner (not in persistence)
    because vbt_runner does `from screener.persistence import read_panel` —
    Python binds `read_panel` as a local module attribute at import time.

    Returns (start_iso, end_iso) derived from the fixture's date endpoints —
    B-2 fix: tests use this dynamic range, not hardcoded 2024-01-01..2024-12-31.
    """
    snapshot_df = _build_perfect_foresight_snapshot(synthetic_ohlcv_panel)

    def _fake_read_panel(_snapshot_date: str | pd.Timestamp) -> pd.DataFrame:
        return synthetic_ohlcv_panel

    def _fake_load_snapshots(
        _start: pd.Timestamp, _end: pd.Timestamp
    ) -> pd.DataFrame:
        return snapshot_df

    monkeypatch.setattr(
        "screener.backtest.vbt_runner.read_panel", _fake_read_panel
    )
    monkeypatch.setattr(
        "screener.backtest.vbt_runner._load_snapshots_in_range", _fake_load_snapshots
    )

    return _fixture_date_range(synthetic_ohlcv_panel)


def test_no_lookahead_correct_path(_patched_runner: tuple[str, str]) -> None:
    """_lookahead=False: harness applies .shift(1); foresight is negated; return stays near random."""
    start, end = _patched_runner
    result = vbt_runner.run(start, end, _lookahead=False)
    assert abs(result.total_return) < LOOKAHEAD_FALSE_MAX_RETURN, (
        f"Look-ahead detected: total_return={result.total_return:+.2%} "
        f"exceeds noise threshold {LOOKAHEAD_FALSE_MAX_RETURN:.0%} "
        f"(range {start}..{end}, {len(result.windows)} windows). "
        f"Check that vbt_runner.run() applies .shift(1, fill_value=False) to entries/exits "
        f"in the `else` branch of `if _lookahead:`. (FND-04 / BCK-02 / D-19)"
    )


def test_no_lookahead_mutation_detected(_patched_runner: tuple[str, str]) -> None:
    """_lookahead=True: harness skips .shift(1); foresight wins big — proves the gate is live."""
    start, end = _patched_runner
    result = vbt_runner.run(start, end, _lookahead=True)
    assert result.total_return > LOOKAHEAD_TRUE_MIN_RETURN, (
        f"Mutation backdoor failed to outperform: total_return={result.total_return:+.2%} "
        f"is below the {LOOKAHEAD_TRUE_MIN_RETURN:.0%} threshold "
        f"(range {start}..{end}, {len(result.windows)} windows). "
        f"Either the perfect-foresight signal construction is wrong, or .shift(1) "
        f"is not actually being bypassed when _lookahead=True. (FND-04 gate broken)"
    )
```
