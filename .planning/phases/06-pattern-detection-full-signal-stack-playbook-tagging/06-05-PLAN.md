```
@app.command("refresh-fundamentals")
def refresh_fundamentals(
    force: Annotated[bool, typer.Option("--force", ...)] = False,
    skip_insider: Annotated[bool, typer.Option("--skip-insider", ...)] = False,
    insider_only: Annotated[bool, typer.Option("--insider-only", ...)] = False,
) -> None:
    """Refresh fundamentals (Finnhub earnings + EPS history + EDGAR Form 4).
    45-day post-quarter-end lag enforced at persistence read (D-13b).
    """
    configure_logging()
    if not insider_only:
        _ensure_edgar_identity()  # only needed when EDGAR step will run; but safe to call always
    today = date.today()
    try:
        if not insider_only:
            from screener.data.fundamentals import refresh_fundamentals as ref_fund
            ref_fund(today=today, force=force)
        if not skip_insider:
            _ensure_edgar_identity()
            from screener.data.insider import refresh_insider as ref_ins
            ref_ins(today=today)
        log.info("refresh_fundamentals_ok", skip_insider=skip_insider, insider_only=insider_only)
    except Exception as e:
        log.error("refresh_fundamentals_failed", error_type=type(e).__name__)
        raise typer.Exit(code=1) from e
```

```
def _add_catalyst_columns(panel: pd.DataFrame, fundamentals: pd.DataFrame, as_of: pd.Timestamp) -> pd.DataFrame:
    """Adds catalyst input columns the composite/snapshot/report all consume:
       - days_to_next_earnings (Int64, nullable; computed from fundamentals.next_earnings_date - as_of)
       - earnings_in_3d_warn (bool; days_to_next_earnings <= 3)
       - crossed_52w_high_within_60d (bool; rolling-60 lookback on high_52w column)
       - insider_cluster_buy (bool; per-ticker membership in persistence.read_insider_cluster_buy())
       - eps_knowable_from (str ISO date; checker W11 — report.py renders
         "Earnings=0 (EPS pending, knowable YYYY-MM-DD)" when canslim_c_passes==False)
    All five are added per-row at the (ticker, snapshot_date) cross-section.
    """
    out = panel.copy()
    # 1. Earnings proximity from fundamentals
    latest_by_ticker = (
        fundamentals.sort_values("fiscal_quarter_end").groupby("ticker").tail(1).set_index("ticker")
        if not fundamentals.empty else pd.DataFrame()
    )
    next_earn_by_ticker = latest_by_ticker["next_earnings_date"] if "next_earnings_date" in latest_by_ticker.columns else pd.Series(dtype="datetime64[ns]")
    knowable_by_ticker = latest_by_ticker["knowable_from"] if "knowable_from" in latest_by_ticker.columns else pd.Series(dtype="datetime64[ns]")
    tickers = out.index.get_level_values("ticker")
    days_to = tickers.map(
        lambda t: (next_earn_by_ticker.get(t, pd.NaT) - as_of).days if t in next_earn_by_ticker.index and pd.notna(next_earn_by_ticker.get(t)) else pd.NA
    )
    out["days_to_next_earnings"] = pd.array(days_to, dtype="Int64")
    out["earnings_in_3d_warn"] = (out["days_to_next_earnings"].fillna(999) <= 3) & (out["days_to_next_earnings"].fillna(-1) >= 0)
    # 1b. eps_knowable_from — ISO date string per ticker (W11). Empty string when fundamentals row missing.
    def _knowable(t: str) -> str:
        if t in knowable_by_ticker.index and pd.notna(knowable_by_ticker.get(t)):
            return pd.Timestamp(knowable_by_ticker.get(t)).date().isoformat()
        return ""
    out["eps_knowable_from"] = tickers.map(_knowable).astype("string")
    # 2. crossed_52w_high_within_60d: True if close hit high_52w within last 60 bars
    # high_52w is already in panel (Phase 4 04-01); compute per-ticker rolling-60 boolean
    near_52w = (out["close"] >= out["high_52w"] * 0.999).fillna(False)
    crossed = near_52w.groupby(level="ticker").rolling(60, min_periods=1).max().droplevel(0).astype(bool)
    out["crossed_52w_high_within_60d"] = crossed.fillna(False)
    # 3. insider_cluster_buy from persistence
    from screener import persistence
    cluster_tickers = persistence.read_insider_cluster_buy(window_days=30, cluster_size=2, dt=5)
    out["insider_cluster_buy"] = tickers.map(lambda t: t in cluster_tickers).astype(bool)
    return out
```

```
def _build_pattern_audit_df(panel: pd.DataFrame, as_of: pd.Timestamp) -> pd.DataFrame:
    """Construct per-leg pattern audit DataFrame for the current cross-section.

    Per D-05: VCP picks emit per-leg rows from diag["legs"]; flag picks emit a
    single row with leg_idx=0. Conforms to PatternAuditSchema (Plan 01).

    Checker B2: VCP rows now read diag["legs"] (added in Plan 02 Task 2) — each
    leg carries real (start_date, end_date, high, low, depth, avg_volume). Older
    zero-filled placeholder version is removed.
    """
    rows = []
    cross = panel.xs(as_of, level="date") if as_of in panel.index.get_level_values("date") else pd.DataFrame()
    from screener.indicators.patterns import decode_pattern_diagnostics
    as_of_ts = pd.Timestamp(as_of)
    for ticker, row in cross.iterrows():
        diag = decode_pattern_diagnostics(row["pattern_diagnostics"])
        if diag["type"] == "vcp":
            legs = diag.get("legs") or []
            if not legs:
                # Plan 02 guarantees `legs` for every VCP — empty here means a
                # downstream regression: skip with a structured-log event so the
                # write doesn't proceed with placeholders.
                log.warning("pattern_audit_vcp_missing_legs", ticker=ticker, snapshot_date=str(as_of))
                continue
            for leg in legs:
                rows.append({
                    "ticker": str(ticker),
                    "snapshot_date": as_of_ts,
                    "pattern_type": "vcp",
                    "leg_idx": int(leg["leg_idx"]),
                    "start_date": pd.Timestamp(leg["start_date"]),
                    "end_date": pd.Timestamp(leg["end_date"]),
                    "high": float(leg["high"]),
                    "low": float(leg["low"]),
                    "depth": float(leg["depth"]),
                    "avg_volume": float(leg["avg_volume"]),
                })
        elif diag["type"] == "flag":
            # Flag picks: single audit row from top-level diag fields (no legs sub-field).
            pivot = float(diag.get("pivot_price", 0.0))
            range_tightness = float(diag.get("range_tightness", 0.0))
            rows.append({
                "ticker": str(ticker),
                "snapshot_date": as_of_ts,
                "pattern_type": "flag",
                "leg_idx": 0,
                "start_date": as_of_ts,
                "end_date": as_of_ts,
                "high": pivot,
                "low": pivot * (1.0 - range_tightness) if range_tightness > 0 else pivot * 0.95,
                "depth": range_tightness if range_tightness > 0 else 0.05,
                "avg_volume": 0.0,  # flag dict does not carry avg_volume; PatternAuditSchema accepts 0
            })
    if not rows:
        return pd.DataFrame(columns=["ticker","snapshot_date","pattern_type","leg_idx","start_date","end_date","high","low","depth","avg_volume"])
    return pd.DataFrame(rows)
```
